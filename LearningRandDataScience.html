如何宣告變數

  • 用<-指定
  • 或用=指定
  • ==用來作邏輯運算子等於
firstVariable <- 'Hello World!'
print(firstVariable)
## [1] "Hello World!"
secondVariable = 2
print(secondVariable)
## [1] 2

如何寫if

revenueCalculator <- function(unitCost=50, unit){
    revenue <- unitCost * unit
    if(unit > 100) {
      revenue <- revenue * 0.8
    }
    return(revenue)
}
revenueCalculator(unit=99)
## [1] 4950
revenueCalculator(unit=101)
## [1] 4040

如何存取不同data type中的元素

vector1 <- seq(1, 9)
vector1[1]
## [1] 1
matrix <- matrix(vector1, nrow=3)
matrix[2,3]
## [1] 8
vector2 <- seq(10, 18)
dataFrame <- data.frame(vector1, vector2)
dataFrame[, 2]
## [1] 10 11 12 13 14 15 16 17 18
dataFrame[2, ]
##   vector1 vector2
## 2       2      11
